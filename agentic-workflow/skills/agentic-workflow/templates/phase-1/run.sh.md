# run.sh.md

Phase 1 (Daily Self-Critique) 오케스트레이터 스크립트 템플릿.

- **생성 경로**: `.github/workflows/daily-critique/run.sh`
- **목적**: 출력물 수집 -> 에이전트 평가 -> 저점수 이슈 생성 -> 메트릭 저장 및 커밋의 전체 흐름을 조율한다.
- **치환 변수**: `{{owner}}`, `{{repo}}`, `{{default_branch}}`, `{{agent_cmd}}`, `{{agent_model}}`, `{{agent_model_flag}}`

---

````sh
#!/bin/sh
set -eu

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
DATE="$(date -u +%Y-%m-%d)"
RESULT_DIR="${SCRIPT_DIR}/results"
METRICS_DIR=".agentic-workflow/metrics"

mkdir -p "${RESULT_DIR}"
mkdir -p "${METRICS_DIR}"

# ── 1. 출력물 수집 ──────────────────────────────────────
echo "[daily-critique] Collecting bot outputs..."
chmod +x "${SCRIPT_DIR}/collect-outputs.sh"
"${SCRIPT_DIR}/collect-outputs.sh" "${RESULT_DIR}"

COLLECTED="${RESULT_DIR}/collected-outputs.json"

if [ ! -s "${COLLECTED}" ] || [ "$(cat "${COLLECTED}")" = "[]" ]; then
  echo "[daily-critique] No bot outputs found for evaluation. Skipping."
  exit 0
fi

# ── 2. 에이전트 평가 실행 ───────────────────────────────
echo "[daily-critique] Running agent critique..."
CRITIQUE_RESULT="${RESULT_DIR}/critique-result.json"

AGENT_MODEL_ARG=""
if [ -n "${AGENT_MODEL:-}" ] && [ -n "${AGENT_MODEL_FLAG:-}" ]; then
  AGENT_MODEL_ARG="${AGENT_MODEL_FLAG} ${AGENT_MODEL}"
fi

# shellcheck disable=SC2086
cat "${COLLECTED}" | ${AGENT_CMD} ${AGENT_MODEL_ARG} \
  --print \
  "${SCRIPT_DIR}/critique-prompt.md" \
  > "${CRITIQUE_RESULT}"

if [ ! -s "${CRITIQUE_RESULT}" ]; then
  echo "[daily-critique] Agent returned empty result. Aborting."
  exit 1
fi

# ── 3. 저점수 항목 이슈 생성 ────────────────────────────
echo "[daily-critique] Checking for low-score items..."

LOW_ITEMS="$(cat "${CRITIQUE_RESULT}" \
  | python3 -c "
import sys, json
data = json.load(sys.stdin)
for item in data.get('low_score_items', []):
    print(item.get('category', ''), item.get('score', ''), item.get('pr_number', ''), item.get('detail', ''), sep='\t')
" 2>/dev/null || true)"

if [ -n "${LOW_ITEMS}" ]; then
  echo "${LOW_ITEMS}" | while IFS='	' read -r category score pr_number detail; do
    [ -z "${category}" ] && continue
    gh issue create \
      --title "[daily-critique] ${category} scored ${score}/100 (PR #${pr_number})" \
      --body "## Daily Self-Critique Alert

**Date**: ${DATE}
**PR**: #${pr_number}
**Category**: ${category}
**Score**: ${score}/100

### Detail
${detail}

---
*Generated by daily-critique workflow.*" \
      --label "agentic-workflow,daily-critique"
    echo "[daily-critique] Created issue for ${category} (score: ${score})"
  done
fi

# ── 4. 메트릭 저장 ──────────────────────────────────────
METRICS_FILE="${METRICS_DIR}/daily-critique-${DATE}.json"
cp "${CRITIQUE_RESULT}" "${METRICS_FILE}"
echo "[daily-critique] Metrics saved to ${METRICS_FILE}"

# ── 5. 결과 커밋 및 푸시 ────────────────────────────────
git config user.name "github-actions[bot]"
git config user.email "github-actions[bot]@users.noreply.github.com"
git add "${METRICS_DIR}/"
if git diff --cached --quiet; then
  echo "[daily-critique] No metric changes to commit."
else
  git commit -m "chore(daily-critique): add metrics for ${DATE}"
  git push origin "{{default_branch}}"
  echo "[daily-critique] Metrics committed and pushed."
fi

# ── 6. 정리 ─────────────────────────────────────────────
rm -rf "${RESULT_DIR}"
echo "[daily-critique] Done."
````
