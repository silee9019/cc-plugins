# run.sh.md

Phase 4 (Weekly Self-Improvement) 오케스트레이터 스크립트 템플릿.

- **생성 경로**: `.github/workflows/weekly-self-improve/run.sh`
- **목적**: 최근 7일 daily-critique 메트릭을 수집하고, 에이전트가 프로젝트 가이드라인을 분석·개선하도록 실행한 뒤, 변경사항이 있으면 PR을 생성한다.
- **치환 변수**: `{{owner}}`, `{{repo}}`, `{{default_branch}}`, `{{agent_cmd}}`, `{{agent_model_flag}}`
- **의존성**: gh CLI, git, `{{agent_cmd}}`

---

````sh
#!/bin/sh
set -eu

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
OWNER="{{owner}}"
REPO="{{repo}}"
DEFAULT_BRANCH="{{default_branch}}"
METRICS_DIR=".agentic-workflow/metrics"
DATE="$(date -u +%Y-%m-%d)"

# ── 1. 최근 7일 메트릭 수집 ──────────────────────────────
echo "[weekly-self-improve] 최근 7일 메트릭 수집 중..."

COLLECTED=""
DAYS_BACK=0
while [ "$DAYS_BACK" -lt 7 ]; do
  # macOS와 GNU date 모두 지원
  if date -v -1d > /dev/null 2>&1; then
    TARGET_DATE=$(date -u -v "-${DAYS_BACK}d" +%Y-%m-%d)
  else
    TARGET_DATE=$(date -u -d "${DAYS_BACK} days ago" +%Y-%m-%d)
  fi
  FILE="${METRICS_DIR}/daily-critique-${TARGET_DATE}.json"
  if [ -f "$FILE" ]; then
    COLLECTED="${COLLECTED} ${FILE}"
  fi
  DAYS_BACK=$((DAYS_BACK + 1))
done

# 공백 제거 후 비어있는지 확인
COLLECTED=$(printf '%s' "$COLLECTED" | sed 's/^ *//')

if [ -z "$COLLECTED" ]; then
  echo "[weekly-self-improve] 평가 데이터가 부족합니다. Phase 1을 먼저 실행하세요."
  exit 0
fi

FILE_COUNT=$(printf '%s' "$COLLECTED" | wc -w | tr -d ' ')
echo "[weekly-self-improve] ${FILE_COUNT}개의 메트릭 파일 발견."

# ── 2. 에이전트 개선 분석 실행 ────────────────────────────
echo "[weekly-self-improve] 에이전트 개선 분석 실행 중..."

AGENT_MODEL_ARG=""
if [ -n "${AGENT_MODEL:-}" ] && [ -n "${AGENT_MODEL_FLAG:-}" ]; then
  AGENT_MODEL_ARG="${AGENT_MODEL_FLAG} ${AGENT_MODEL}"
fi

# 메트릭 파일들을 하나의 JSON 배열로 합침
printf '[\n' > /tmp/metrics-combined.json
FIRST=1
for f in $COLLECTED; do
  if [ "$FIRST" -eq 1 ]; then
    FIRST=0
  else
    printf ',\n' >> /tmp/metrics-combined.json
  fi
  cat "$f" >> /tmp/metrics-combined.json
done
printf '\n]\n' >> /tmp/metrics-combined.json

# shellcheck disable=SC2086
cat /tmp/metrics-combined.json | ${AGENT_CMD} ${AGENT_MODEL_ARG} \
  --prompt-file "${SCRIPT_DIR}/improve-prompt.md"

# ── 3. 변경사항 확인 및 PR 생성 ──────────────────────────
echo "[weekly-self-improve] 변경사항 확인 중..."

git add -A
if git diff --cached --quiet; then
  echo "[weekly-self-improve] 이번 주 개선 사항이 없습니다."
  exit 0
fi

BRANCH="chore/self-improve-${DATE}"
git config user.name "github-actions[bot]"
git config user.email "github-actions[bot]@users.noreply.github.com"
git checkout -b "$BRANCH"
git commit -m "chore(self-improve): weekly guideline improvements ${DATE}"
git push -u origin "$BRANCH"

CHANGED_FILES=$(git diff --name-only "origin/${DEFAULT_BRANCH}...HEAD")

gh pr create --repo "${OWNER}/${REPO}" \
  --base "$DEFAULT_BRANCH" --head "$BRANCH" \
  --title "chore: weekly self-improvement ${DATE}" \
  --body "$(printf '## Weekly Self-Improvement\n\n**기간**: %s ~ %s\n**분석 메트릭**: %s일분\n\n### 변경 파일\n```\n%s\n```\n\n---\n*Generated by weekly-self-improve workflow.*' \
    "$(date -u -d '7 days ago' +%Y-%m-%d 2>/dev/null || date -u -v '-7d' +%Y-%m-%d)" \
    "$DATE" "$FILE_COUNT" "$CHANGED_FILES")"

echo "[weekly-self-improve] PR 생성 완료."

# ── 4. 정리 ──────────────────────────────────────────────
rm -f /tmp/metrics-combined.json
echo "[weekly-self-improve] 완료."
````
